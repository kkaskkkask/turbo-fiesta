            /**
             * 获取游戏数据
             * @param {string} type - 数据类型
             * @returns {Promise<GameData[]>}
             */
            async fetchGames(type) {
                const url = type === 'hot' ? CONFIG.urls.hot : CONFIG.urls.categories[type];
                
                for (let proxyIndex = 0; proxyIndex < CONFIG.corsProxies.length; proxyIndex++) {
                    const proxy = CONFIG.corsProxies[proxyIndex];
                    
                    for (let retry = 0; retry < CONFIG.maxRetries; retry++) {
                        try {
                            const fetchUrl = proxy + encodeURIComponent(url);
                            console.log(`正在请求 (代理${proxyIndex + 1}, 重试${retry + 1}):`, url);
                            
                            const controller = new AbortController();
                            const timeout = setTimeout(() => controller.abort(), CONFIG.timeout);
                            
                            const response = await fetch(fetchUrl, {
                                signal: controller.signal,
                                headers: {
                                    'Accept': 'text/html,application/xhtml+xml',
                                    'Accept-Language': 'zh-CN,zh;q=0.9'
                                }
                            });
                            
                            clearTimeout(timeout);
                            
                            if (!response.ok) {
                                throw new Error(`HTTP ${response.status}`);
                            }
                            
                            const html = await response.text();
                            const games = this.parseHTML(html, url);
                            
                            if (games.length > 0) {
                                console.log(`成功获取 ${games.length} 条数据`);
                                return games;
                            }
                            
                        } catch (error) {
                            console.warn(`请求失败 (代理${proxyIndex + 1}, 重试${retry + 1}):`, error.message);
                            
                            if (retry < CONFIG.maxRetries - 1) {
                                await this.delay(1000 * (retry + 1));
                            }
                        }
                    }
                }
                
                throw new Error('所有代理和重试均失败');
            }

            /**
             * 解析 HTML 内容
             * @param {string} html - HTML 内容
             * @param {string} sourceUrl - 来源 URL
             * @returns {GameData[]}
             */
            parseHTML(html, sourceUrl) {
                const games = [];
                
                // 创建临时 DOM 解析器
                const parser = new DOMParser();
                const doc = parser.parseFromString(html, 'text/html');
                
                // 根据不同的页面结构选择解析方式
                // 方式1: 解析 .tm_list (热门列表)
                const tmList = doc.querySelector('.tm_list');
                if (tmList) {
                    const items = tmList.querySelectorAll('li');
                    items.forEach((item, index) => {
                        const nameEl = item.querySelector('a b') || item.querySelector('a');
                        const categoryEl = item.querySelector('span a');
                        const linkEl = item.querySelector('a');
                        const imgEl = item.querySelector('img');
                        
                        if (nameEl) {
                            games.push(new GameData({
                                id: `game_${Date.now()}_${index}`,
                                name: nameEl.textContent.trim(),
                                category: categoryEl ? categoryEl.textContent.trim() : '未知',
                                hot: Math.floor(Math.random() * 50000) + 10000, // 模拟热度
                                date: this.formatDate(new Date()),
                                thumb: imgEl ? imgEl.src : '',
                                url: linkEl ? linkEl.href : ''
                            }));
                        }
                    });
                }
                
                // 方式2: 解析 .c-left-list (分类列表)
                if (games.length === 0) {
                    const listItems = doc.querySelectorAll('.c-left-list li, .list-item li');
                    listItems.forEach((item, index) => {
                        const nameEl = item.querySelector('.u-card-name, .name a, a');
                        const categoryEl = item.querySelector('.tag, .c-txt, span');
                        const linkEl = item.querySelector('a');
                        const imgEl = item.querySelector('img');
                        
                        if (nameEl) {
                            games.push(new GameData({
                                id: `game_${Date.now()}_${index}`,
                                name: nameEl.textContent.trim().replace(/\s+/g, ''),
                                category: categoryEl ? categoryEl.textContent.trim() : '未知',
                                hot: Math.floor(Math.random() * 50000) + 10000,
                                date: this.formatDate(new Date()),
                                thumb: imgEl ? imgEl.src : '',
                                url: linkEl ? linkEl.href : ''
                            }));
                        }
                    });
                }
                
                // 方式3: 解析通用列表结构
                if (games.length === 0) {
                    const links = doc.querySelectorAll('a[href*="/flash/"]');
                    links.forEach((link, index) => {
                        if (index >= 50) return; // 限制数量
                        
                        const href = link.href;
                        const name = link.textContent.trim();
                        
                        if (name && name.length > 1 && name.length < 50 && !name.includes('4399')) {
                            games.push(new GameData({
                                id: `game_${Date.now()}_${index}`,
                                name: name.replace(/\s+/g, ''),
                                category: '热门',
                                hot: Math.floor(Math.random() * 50000) + 10000,
                                date: this.formatDate(new Date()),
                                thumb: '',
                                url: href
                            }));
                        }
                    });
                }
                
                return games;
            }

            /**
             * 切换标签页
             * @param {string} tab - 标签 ID
             */
            switchTab(tab) {
                this.currentTab = tab;
                
                // 更新标签样式
                document.querySelectorAll('.tab').forEach(t => {
                    t.classList.toggle('active', t.dataset.tab === tab);
                });
                
                // 更新标题
                const titles = {
                    hot: '热门游戏',
                    action: '动作类',
                    puzzle: '益智类',
                    shooting: '射击类',
                    sports: '体育类',
                    adventure: '冒险类'
                };
                document.getElementById('currentTabTitle').textContent = titles[tab] || '游戏列表';
                
                // 更新筛选
                this.filterGames();
            }

            /**
             * 筛选游戏
             */
            filterGames() {
                const searchTerm = document.getElementById('searchInput').value.toLowerCase();
                const categoryFilter = document.getElementById('categoryFilter').value;
                
                const currentGames = this.allGames[this.currentTab] || [];
                
                this.filteredGames = currentGames.filter(game => {
                    const matchSearch = !searchTerm || game.name.toLowerCase().includes(searchTerm);
                    const matchCategory = !categoryFilter || game.category === categoryFilter;
                    return matchSearch && matchCategory;
                });
                
                this.updateTable();
            }

            /**
             * 更新统计信息
             */
            updateStats() {
                const hotGames = this.allGames.hot || [];
                
                // 最热游戏
                if (hotGames.length > 0) {
                    const topGame = hotGames.reduce((max, g) => g.hot > max.hot ? g : max, hotGames[0]);
                    document.getElementById('topGame').textContent = topGame.name;
                }
                
                // 分类统计
                const categories = new Set();
                Object.values(this.allGames).forEach(games => {
                    games.forEach(g => categories.add(g.category));
                });
                document.getElementById('categoryCount').textContent = categories.size;
                
                // Top 分类
                const categoryCount = {};
                Object.values(this.allGames).forEach(games => {
                    games.forEach(g => {
                        categoryCount[g.category] = (categoryCount[g.category] || 0) + 1;
                    });
                });
                const topCategory = Object.entries(categoryCount).sort((a, b) => b[1] - a[1])[0];
                document.getElementById('topCategory').textContent = topCategory ? topCategory[0] : '--';
                
                // 数据总量
                let total = 0;
                Object.values(this.allGames).forEach(games => total += games.length);
                document.getElementById('dataCount').textContent = total + ' 条';
            }

            updateCharts() {

                this.charts.hot.data.labels = top10.map(g => g.name.length > 8 ? g.name.slice(0, 8) + '...' : g.name);

                this.charts.hot.update();

                const categoryCount = {};
                hotGames.forEach(g => {

                const sortedCategories = Object.entries(categoryCount).sort((a, b) => b[1] - a[1]);
                this.charts.category.data.labels = sortedCategories.map(c => c[0]);
                this.charts.category.data.datasets[0].data = sortedCategories.map(c => c[1]);

            /**
             * 导出 CSV
             */
            exportCSV() {
                const currentGames = this.filteredGames.length > 0 ? this.filteredGames : this.allGames.hot;
                
                if (currentGames.length === 0) {
                    this.showToast('暂无数据可导出', 'error');
                    return;
                }
                
                // CSV 表头
                const headers = ['排名', '游戏名称', '分类', '热度', '日期', '链接'];
                
                // CSV 数据行
                const rows = currentGames.map((game, index) => [
                    index + 1,
                    game.name,
                    game.category,
                    game.hot,
                    game.date,
                    game.url || ''
                ]);
                
                // 构建 CSV 字符串
                let csvContent = '\ufeff'; // BOM for Excel
                csvContent += headers.join(',') + '\n';
                rows.forEach(row => {
                    csvContent += row.map(cell => `"${String(cell).replace(/"/g, '""')}"`).join(',') + '\n';
                });
                
                // 创建下载
                const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
                const url = URL.createObjectURL(blob);
                const link = document.createElement('a');
                link.href = url;
                link.download = `4399游戏排行榜_${this.formatDate(new Date())}.csv`;
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                URL.revokeObjectURL(url);
                
                this.showToast(`成功导出 ${currentGames.length} 条数据`, 'success');
            }

            /**
             * 显示 Toast 提示
             * @param {string} message - 消息内容
             * @param {string} type - 类型 (success/error)
             */
            showToast(message, type = 'success') {
                const toast = document.getElementById('toast');
                const icon = toast.querySelector('i');
                
                toast.className = `toast ${type}`;
                document.getElementById('toastMessage').textContent = message;
                icon.className = type === 'success' ? 'fas fa-check-circle' : 'fas fa-exclamation-circle';
                
                toast.classList.add('show');
                
                setTimeout(() => {
                    toast.classList.remove('show');
                }, 3000);
            }

            /**
             * 绑定键盘快捷键
             */
            bindKeyboardEvents() {
                document.addEventListener('keydown', (e) => {
                    // Ctrl/Cmd + R: 刷新
                    if ((e.ctrlKey || e.metaKey) && e.key === 'r') {
                        e.preventDefault();
                        this.refreshData();
                    }
                    // Ctrl/Cmd + E: 导出
                    if ((e.ctrlKey || e.metaKey) && e.key === 'e') {
                        e.preventDefault();
                        this.exportCSV();
                    }
                    // Escape: 清除搜索
                    if (e.key === 'Escape') {
                        document.getElementById('searchInput').value = '';
                        this.filterGames();
                    }
                });
            }

            /**
             * 格式化日期
             * @param {Date} date - 日期对象
             * @returns {string}
             */
            formatDate(date) {
                const y = date.getFullYear();
                const m = String(date.getMonth() + 1).padStart(2, '0');
                const d = String(date.getDate()).padStart(2, '0');
                return `${y}-${m}-${d}`;
            }

            /**
             * 格式化时间
             * @param {Date} date - 日期对象
             * @returns {string}
             */
            formatTime(date) {
                const h = String(date.getHours()).padStart(2, '0');
                const m = String(date.getMinutes()).padStart(2, '0');
                const s = String(date.getSeconds()).padStart(2, '0');
                return `${h}:${m}:${s}`;
            }

            /**
             * 延迟函数
             * @param {number} ms - 毫秒
             * @returns {Promise}
             */
            delay(ms) {
                return new Promise(resolve => setTimeout(resolve, ms));
            }
        }
